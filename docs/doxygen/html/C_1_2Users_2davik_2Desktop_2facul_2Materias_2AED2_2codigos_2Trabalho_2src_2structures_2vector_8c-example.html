Cria um novo vetor dinâmico.

Cria um novo vetor dinâmico

Parâmetros
initial_capacityCapacidade inicial do vetor
Retorna
Ponteiro para o vetor criado ou NULL se falhar

Aloca e inicializa uma nova estrutura Vector com a capacidade inicial especificada. O vetor pode ser expandido dinamicamente conforme necessário.

A função:

  • Aloca memória para a estrutura Vector
  • Aloca um array de WordInfo com a capacidade inicial
  • Inicializa size como 0 e capacity como initial_capacity
Nota
Se qualquer alocação falhar, a função libera a memória já alocada e retorna NULL para evitar vazamentos de memória.
O vetor começa vazio (size = 0) mas com capacidade para initial_capacity elementos.
Aviso
É responsabilidade do chamador verificar se o retorno é NULL e tratar adequadamente falhas de alocação.
Vector *vec = create_vector(100);
if (vec == NULL) {
fprintf(stderr, "Erro ao criar vetor.\n");
return 1;
}

Vector* create_vector(int initial_capacity) {
Vector *vec = (Vector*)malloc(sizeof(Vector));
if (vec == NULL) return NULL;
vec->words = (WordInfo*)malloc(initial_capacity * sizeof(WordInfo));
if (vec->words == NULL) {
free(vec);
return NULL;
}
vec->size = 0;
vec->capacity = initial_capacity;
return vec;
}
void expand_vector(Vector *vec) {
int new_capacity = vec->capacity * 2;
WordInfo *new_words = (WordInfo*)realloc(vec->words, new_capacity * sizeof(WordInfo));
if (new_words != NULL) {
vec->words = new_words;
vec->capacity = new_capacity;
}
}
void insert_word_vector(Vector *vec, const char *word, const char *title,
const char *author, const char *verse_snippet, int freq_in_song) {
if (vec == NULL) return;
for (int i = 0; i < vec->size; i++) {
if (strcmp(vec->words[i].word, word) == 0) {
vec->words[i].total_frequency += freq_in_song;
if (freq_in_song > vec->words[i].frequency_in_song) {
if (vec->words[i].title) free((void*)vec->words[i].title);
if (vec->words[i].author) free((void*)vec->words[i].author);
if (vec->words[i].verse_snippet) free((void*)vec->words[i].verse_snippet);
vec->words[i].title = strdup(title);
vec->words[i].author = strdup(author);
vec->words[i].verse_snippet = strdup(verse_snippet);
vec->words[i].frequency_in_song = freq_in_song;
}
return;
}
}
if (vec->size >= vec->capacity) {
expand_vector(vec);
}
vec->words[vec->size].word = strdup(word);
vec->words[vec->size].title = strdup(title);
vec->words[vec->size].author = strdup(author);
vec->words[vec->size].verse_snippet = strdup(verse_snippet);
vec->words[vec->size].frequency_in_song = freq_in_song;
vec->words[vec->size].total_frequency = freq_in_song;
vec->size++;
}
WordInfo* search_word_vector(Vector *vec, const char *word) {
if (vec == NULL || word == NULL) return NULL;
int left = 0;
int right = vec->size - 1;
while (left <= right) {
int mid = left + (right - left) / 2;
int comparison = strcmp(vec->words[mid].word, word);
if (comparison == 0) {
return &vec->words[mid];
} else if (comparison < 0) {
left = mid + 1;
} else {
right = mid - 1;
}
}
return NULL;
}
int compare_words(const void *a, const void *b) {
return strcmp(((WordInfo*)a)->word, ((WordInfo*)b)->word);
}
void sort_vector(Vector *vec) {
if (vec == NULL) return;
qsort(vec->words, vec->size, sizeof(WordInfo), compare_words);
}
void free_vector(Vector *vec) {
if (vec == NULL) return;
for (int i = 0; i < vec->size; i++) {
if (vec->words[i].word) free(vec->words[i].word);
if (vec->words[i].title) free(vec->words[i].title);
if (vec->words[i].author) free(vec->words[i].author);
if (vec->words[i].verse_snippet) free(vec->words[i].verse_snippet);
}
free(vec->words);
free(vec);
}
