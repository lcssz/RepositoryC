160 int comparison = strcmp(new_node->word, current->word);
161
162 if (comparison < 0) {
163 if (current->left == NULL) {
164 current->left = new_node;

166 insert_node_recursive(current->left, new_node);
167 }
168 } else if (comparison > 0) {

273void insert_node_complete_bst(TreeComplete *tree, WordInfo *info) {

275
276 NodeComplete *new_node = create_node_complete(info);
277 if (new_node == NULL) return;

280 tree->root = new_node;
281 return;

284 NodeComplete *current = tree->root;
285 NodeComplete *parent = NULL;
286
287 while (current != NULL) {
288 parent = current;
289 int comparison = strcmp(info->word, current->info->word);

291 if (comparison < 0) {
292 current = current->left;
293 } else if (comparison > 0) {
294 current = current->right;
295 } else {
296
297 current->info->total_frequency += info->total_frequency;
298 if (info->frequency_in_song > current->info->frequency_in_song) {
299 current->info->title = info->title;

303 }
304 free(new_node);
305 return;
306 }
307 }
308
309 int comparison = strcmp(info->word, parent->info->word);
310 if (comparison < 0) {
311 parent->left = new_node;
