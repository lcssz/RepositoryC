59char* clean_word(const char *word) {
60 if (word == NULL) return NULL;
61
62 char *cleaned = strdup(word);
63 if (cleaned == NULL) return NULL;
64
65
66 while (*cleaned && !isalpha(*cleaned)) {
67 memmove(cleaned, cleaned + 1, strlen(cleaned));
68 }
69
70
71 int len = strlen(cleaned);
72 while (len > 0 && !isalpha(cleaned[len - 1])) {
73 cleaned[--len] = '\0';
74 }
75
76
77 for (int i = 0; cleaned[i]; i++) {
78 cleaned[i] = tolower(cleaned[i]);
79 }
80
81 return cleaned;
82}

194void process_music_file_complete(const char *filepath, Repository *repo) {
195 FILE *file = fopen(filepath, "r");
196 if (file == NULL) {
197 perror("Erro ao abrir o arquivo");
198 return;
199 }
200
201 char line_buffer[1024];
202 char title[256] = "";
203 char author[256] = "";
204 char current_verse[1024] = "";
205
206
207 if (fgets(line_buffer, sizeof(line_buffer), file) != NULL) {
208 line_buffer[strcspn(line_buffer, "\n")] = '\0';
209 strcpy(title, line_buffer);
210 }
211
212
213 if (fgets(line_buffer, sizeof(line_buffer), file) != NULL) {
214 line_buffer[strcspn(line_buffer, "\n")] = '\0';
215 strcpy(author, line_buffer);
216 }
217
218
219 while (fgets(line_buffer, sizeof(line_buffer), file) != NULL) {
220 line_buffer[strcspn(line_buffer, "\n")] = '\0';
221 strcpy(current_verse, line_buffer);
222
223 char *line_copy = strdup(line_buffer);
224 if (line_copy == NULL) continue;
225
226 char *token = strtok(line_copy, " \t\n");
227 while (token != NULL) {
228 char *cleaned_word = clean_word(token);
229 if (cleaned_word != NULL && strlen(cleaned_word) >= 4) {
230
231 WordInfo *info = malloc(sizeof(WordInfo));
232 if (info != NULL) {
233 info->word = strdup(cleaned_word);
234 info->title = strdup(title);
235 info->author = strdup(author);
236 info->verse_snippet = strdup(current_verse);
237 info->frequency_in_song = 1;
238 info->total_frequency = 1;
239
240
241 insert_word_vector(repo->word_vector, cleaned_word, title, author, current_verse, 1);
242 insert_node_complete_bst(repo->bst_complete, info);
243 insert_node_complete_avl(repo->avl_complete, info);
244 }
245 }
246 free(cleaned_word);
247 token = strtok(NULL, " \t\n");
248 }
249 free(line_copy);
250 }
251
252 fclose(file);
253}

285void search_word_in_all_structures(Repository *repo, const char *word) {
286 if (repo == NULL || word == NULL) return;
287
288 char *cleaned_word = clean_word(word);
289 if (cleaned_word == NULL) return;
290
291 clock_t start, end;
292 double vector_time, bst_time, avl_time;
293
294
295 start = clock();
296 WordInfo *vector_result = search_word_vector(repo->word_vector, cleaned_word);
297 end = clock();
298 vector_time = ((double)(end - start)) / CLOCKS_PER_SEC;
299
300
301 start = clock();
302 NodeComplete *bst_result = search_tree_complete(repo->bst_complete->root, cleaned_word);
303 end = clock();
304 bst_time = ((double)(end - start)) / CLOCKS_PER_SEC;
305
306
307 start = clock();
308 NodeComplete *avl_result = search_tree_complete(repo->avl_complete->root, cleaned_word);
309 end = clock();
310 avl_time = ((double)(end - start)) / CLOCKS_PER_SEC;
311
312
313 print_search_results(cleaned_word, vector_result,
314 bst_result ? bst_result->info : NULL,
315 avl_result ? avl_result->info : NULL,
316 vector_time, bst_time, avl_time);
317
318 free(cleaned_word);
319}
