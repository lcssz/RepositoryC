210WordInfo* search_word_vector(Vector *vec, const char *word) {
211 if (vec == NULL || word == NULL) return NULL;
212
213 int left = 0;
214 int right = vec->size - 1;
215
216 while (left <= right) {
217 int mid = left + (right - left) / 2;
218 int comparison = strcmp(vec->words[mid].word, word);
219
220 if (comparison == 0) {
221 return &vec->words[mid];
222 } else if (comparison < 0) {
223 left = mid + 1;
224 } else {
225 right = mid - 1;
226 }
227 }
228
229 return NULL;
230}
